## üìä **Data Management**

### **Profile Data Structure**
```typescript
interface ProfileData {
  id: string;
  firstName: string;
  lastName: string;
  email: string;
  profileImage?: string;
  bio?: string;
  location?: string;
  website?: string;
  dateOfBirth?: string;
  gender?: 'male' | 'female' | 'other' | 'prefer-not-to-say';
  preferences: {
    theme: 'light' | 'dark' | 'auto';
    language: string;
    notifications: { email: boolean; push: boolean; marketing: boolean };
    privacy: { profileVisibility: 'public' | 'private' | 'friends'; showEmail: boolean; showLocation: boolean };
  };
  moviePreferences: {
    favoriteGenres: string[];
    favoriteActors: string[];
    favoriteDirectors: string[];
    preferredLanguages: string[];
    contentRating: 'G' | 'PG' | 'PG-13' | 'R' | 'NC-17' | 'all';
  };
  socialLinks: {
    twitter?: string;
    instagram?: string;
    facebook?: string;
    linkedin?: string;
  };
  statistics: {
    totalMoviesWatched: number;
    totalMoviesRated: number;
    averageRating: number;
    favoriteGenre: string;
    joinDate: string;
    lastActive: string;
  };
}
```
